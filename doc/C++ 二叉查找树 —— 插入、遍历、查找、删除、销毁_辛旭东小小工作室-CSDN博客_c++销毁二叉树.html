二叉查找树的特性：

二叉查找树（Binary Search Tree）（又：二叉搜索树，二叉排序树）

具有以下特性：

（1）若“左子树”不为空，则“左子树”上所有节点的值均小于“根节点”的值；

（2）若“右子树”不为空，则“右子树”上所有节点的值均大于“根节点”的值；

（3）左、右子树也分别为二叉排序树。

  133. BSTNode *BSTree::successor(BSTNode *x)  //找节点(x)的后继节点，也就是该节点的右子树中的最小节点
  134. {
  135. 	BSTNode *y = NULL;
  136. 	if (x->right != NULL)
  137. 	{
  138. 		return minimum(x->right);  
  139. 	}
  140.  
  141. // 如果x没有右子节点。则x有以下两种可能：
  142. // （1） x是"一个左子节点"，则"x的后继节点"为 "它的父节点"。
  143. // （2） x是"一个右子节点"，则查找"x的最低的父节点，并且该父节点要具有左子节点"，找到的这个"最低的父节点"就是"x的后继节点"。
  144.  
  145. 	y  = x->parent;
  146. 	while (y != NULL && x == y->right)
  147. 	{
  148. 		x = y;
  149. 		y = y->parent;
  150. 	}
  151. 	return y;
  152. }

  155. BSTNode *BSTree::predecessor(BSTNode *x)  //找节点(x)的前驱节点是该节点的左子树中的最大节点。
  156. {
  157. 	BSTNode *y = NULL;
  158. 	if (x->left != NULL)
  159. 	{
  160. 		return maximum(x->left);
  161. 	}
  162.  
  163. // 如果x没有左子节点。则x有以下两种可能：
  164. //（1）x是"一个右子节点"，则"x的前驱节点"为 "它的父节点"。
  165. //（2）x是"一个左子节点"，则查找"x的最低的父节点，并且该父节点要具有右子节点"，找到的这个"最低的父节点"就是"x的前驱节点"。
  166.  
  167. 	y = x->parent;
  168. 	while (y != NULL && x == y->left)
  169. 	{
  170. 		x = y;
  171. 		y = y->parent;
  172. 	}
  173. 	return y;
  174. }

  176.  
  177. void BSTree::insert(BSTNode *&tree, BSTNode *z)  // 将节点(z)插入到二叉树(tree)中
  178. {
  179. 	BSTNode *y = NULL;
  180. 	BSTNode *x = tree;
  181.  
  182. 	while (x != NULL)  // 查找z的插入位置
  183. 	{
  184. 		y = x;
  185. 		if (z->key < x->key)
  186. 		{
  187. 			x = x->left;
  188. 		}
  189. 		else
  190. 		{
  191. 			x = x->right;
  192. 		}
  193. 	}
  194.  
  195. 	z->parent = y;
  196. 	if (y == NULL)
  197. 	{
  198. 		tree = z;
  199. 	}
  200. 	else
  201. 		if (z->key < y->key)
  202. 		{
  203. 			y->left = z;
  204. 		}
  205. 		else
  206. 		{
  207. 			y->right = z;
  208. 		}
  209. }

  211.  
  212. BSTNode *BSTree::remove(BSTNode *tree, BSTNode *z)  // 删除二叉树(tree)中的节点(z)，并返回被删除的节点
  213. {
  214. 	BSTNode *x = NULL;
  215. 	BSTNode *y = NULL;
  216.  
  217. 	if (z->left == NULL || z->right == NULL)
  218. 	{
  219. 		y = z;
  220. 	}
  221. 	else
  222. 	{
  223. 		y = successor(z);
  224. 	}
  225.  
  226. 	if (y->left != NULL)
  227. 	{
  228. 		x = y->left;
  229. 	}
  230. 	else
  231. 	{
  232. 		x = y->right;
  233. 	}
  234.  
  235. 	if (x != NULL)
  236. 	{
  237. 		x->parent = y->parent;
  238. 	}
  239.  
  240. 	if (y->parent == NULL)
  241. 	{
  242. 		tree = x;
  243. 	}
  244. 	else
  245. 		if (y == y->parent->left)
  246. 		{
  247. 			y->parent->left = x;
  248. 		}
  249. 		else
  250. 		{
  251. 			y->parent->right = x;
  252. 		}
  253.  
  254. 	if (y != z)
  255. 	{
  256. 		z->key = y->key;
  257. 	}
  258. 	return y;
  259. }
  260.  
  261.  
  262. void BSTree::remove(int key)  // 删除二叉树(tree)中的节点(z)，并返回被删除的节点
  263. {
  264. 	BSTNode *z, *node;
  265. 	z = IteratorSearch(root, key);  //根据给定的key，查找树中是否存在key节点
  266. 	if (z != NULL)
  267. 	{
  268. 		node = remove(root, z);  //传入树根以及待删除的节点(z)
  269. 		if (node != NULL)
  270. 		{
  271. 			delete node;
  272. 		}
  273. 	}
  274. }
  275.  
  276.  
  277. void BSTree::destroy(BSTNode *&tree)  //销毁二叉树
  278. {
  279. 	if (tree == NULL)
  280. 	{
  281. 		return;  //停止函数的执行
  282. 	}
  283. 	if (tree->left != NULL)
  284. 	{
  285. 		return destroy(tree->left);
  286. 	}
  287. 	if (tree->right != NULL)
  288. 	{
  289. 		return destroy(tree->right);
  290. 	}
  291. 	delete tree;
  292. 	tree = NULL;
  293. }
  294.  
  295.  
  296. void BSTree::destroy()  //销毁二叉树
  297. {
  298. 	destroy(root);
  299. }
